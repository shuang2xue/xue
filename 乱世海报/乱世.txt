        // 倒计时功能
        function updateCountdown() {
            const days = document.getElementById('days');
            const hours = document.getElementById('hours');
            const minutes = document.getElementById('minutes');
            const seconds = document.getElementById('seconds');
            
            // 这里只是一个静态示例，实际应用中应计算真实时间差
            let sec = parseInt(seconds.textContent);
            let min = parseInt(minutes.textContent);
            let hrs = parseInt(hours.textContent);
            let dys = parseInt(days.textContent);
            
            sec--;
            
            if (sec < 0) {
                sec = 59;
                min--;
                
                if (min < 0) {
                    min = 59;
                    hrs--;
                    
                    if (hrs < 0) {
                        hrs = 23;
                        dys--;
                        
                        if (dys < 0) {
                            // 倒计时结束
                            dys = hrs = min = sec = 0;
                        }
                    }
                }
            }
            
            // 更新显示
            days.textContent = dys.toString().padStart(2, '0');
            hours.textContent = hrs.toString().padStart(2, '0');
            minutes.textContent = min.toString().padStart(2, '0');
            seconds.textContent = sec.toString().padStart(2, '0');
        }
        
        // 每秒钟更新一次倒计时
        setInterval(updateCountdown, 1000);

            confetti.style.width = '10px';
            confetti.style.height = '10px';
            confetti.style.background = getRandomColor();
            confetti.style.borderRadius = '50%';
            confetti.style.left = `${Math.random() * 100}vw`;
            confetti.style.top = '0';
            confetti.style.zIndex = '9999';
            confetti.style.pointerEvents = 'none';
            document.body.appendChild(confetti);
            
            // 动画
            const animation = confetti.animate([
                { transform: 'translateY(0) rotate(0deg)', opacity: 1 },
                { transform: `translateY(${window.innerHeight}px) rotate(${Math.random() * 360}deg)`, opacity: 0 }
            ], {
                duration: 2000 + Math.random() * 3000,
                easing: 'cubic-bezier(0.1, 0.8, 0.2, 1)'
            });
            
            // 动画结束后移除元素
            animation.onfinish = () => confetti.remove();
        }
        
        function getRandomColor() {
            const colors = ['#ff2070', '#ff8a00', '#a67cff', '#4dccff', '#ffcc00'];
            return colors[Math.floor(Math.random() * colors.length)];
        }
    </script>
